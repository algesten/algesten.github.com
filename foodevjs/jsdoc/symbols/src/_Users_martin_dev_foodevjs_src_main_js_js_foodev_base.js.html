  3  * The Foodev namespace and fundamental methods to create packages/classes.<br/><br/>
  4  * The global foodev object itself is an instance of {@link Package} with additional static method
  5  * as detailed below.<br/>
  6  *
  7  * @see Class
  8  * @see Package
  9  * @namespace
 10  *
 11  * @borrows Package#Class as Class
 12  * @borrows Package#Package as Package
 13  * @borrows Package#RootPackage as RootPackage
 14  *
 15  * @author Martin Algesten
 16  */
 17 var foodev = foodev || {};

 21     // This is here to fool JSLint, this is overwritten with $Package after that
 22     // has been properly declared.
 23     var Package = function () {
 24     };
 25 
 26     /**
 27      * <i><b>This is not a class - the functions documented here exist on all packages
 28      * created using {@link foodev.Package} and {@link foodev.RootPackage}.</b></i>
 29      *
 30      * @name Package
 31      * @constructor
 32      */
 33     var $Package = function (owner, name) {
 34 
 35         /**
 36          * Creates a new class in the current package scope. Calls to {@link foodev.Class} are
 37          * really calling this method.
 38          *
 39          * @param {String} pack The package to create a class in.
 40          * @param {String} cname The class name to create in the package.
 41          * @param {Object} obj The object methods and properties to extend into the new class.
 42          */
 43         var makeClass = function (pack, cname, obj) {
 44 
 45             if (!(pack instanceof Package)) {
 46                 throw new Error('package not created using foodev.Package()');
 47             }
 48 
 49             if (typeof obj !== 'object') {
 50                 throw new Error('obj argument must be an object');
 51             }
 52 
 53             // deliberate ucase Ext to fool jslint on below new.
 54             var Ext = obj.Extends;
 55             var prot = {};
 56 
 57             /**
 58              * <i><b>This is not a class - the functions documented here exist on all classes
 59              * created using {@link foodev.Class}. </b></i>
 60              *
 61              * @name Class
 62              * @constructor
 63              */
 64             var newC = function () {
 65                 if (this.initialize) {
 66                     this.initialize.apply(this, arguments);
 67                 }
 68             };
 69 
 70             if (Ext) {
 71                 if (typeof Ext !== 'function') {
 72                     throw new Error('Extended class is not a function!');
 73                 }
 74                 // reset initializer to empty function to not get
 75                 // any side effects when instantiating new prototype
 76                 var init = Ext.prototype.initialize;
 77                 Ext.prototype.initialize = function () {
 78                 };
 79                 try {
 80                     prot = new Ext();
 81                 } catch (err) {
 82                     throw new Error('Failed instantiate superclass (Extends) for: ' + cname);
 83                 }
 84                 // now restore the initializer.
 85                 if (init === undefined) {
 86                     delete Ext.prototype.initialize;
 87                 } else {
 88                     Ext.prototype.initialize = init;
 89                 }
 90                 obj.Extends = Ext.prototype;
 91             }
 92 
 93             // add incoming functions to prototype
 94             foodev.extend(prot, obj);
 95 
 96             // set the prototype
 97             newC.prototype = prot;
 98 
 99             /**
100              * Returns the owning package of the class.
101              *
102              * <p>
103              * <i>This method exist on all classes created using the {@link foodev.Class}
104              * method.</i>
105              * </p>
106              *
107              * @return {Package} The owning package object.
108              *
109              * @name Class#getPackage
110              * @function
111              */
112             newC.prototype.getPackage = function () {
113                 if (obj.Private) {
114                     return null;
115                 }
116                 return pack;
117             };
118             newC.getPackage = newC.prototype.getPackage;
119 
120             /**
121              * Returns the (local) class name.<br/>
122              * For full class name see {@link Class#getFullClassName}
123              *
124              * <p>
125              * <i>This method exist on all classes created using the {@link foodev.Class}
126              * method.</i>
127              * </p>
128              *
129              * @example
130              *   var point = new foodev.bar.Point();
131              *   point.getClassName(); // 'Point'
132              *
133              * @return {String} The local class name.
134              *
135              * @name Class#getClassName
136              * @function
137              */
138             newC.prototype.getClassName = function () {
139                 return cname;
140             };
141             newC.getClassName = newC.prototype.getClassName;
142             if (!newC.prototype.hasOwnProperty('toString') && prot && prot.toString === Object.toString) {
143                 newC.prototype.toString = newC.prototype.getClassName;
144             }
145             /**
146              * Returns the full class name.
147              *
148              * <p>
149              * <i>This method exist on all classes created using the {@link foodev.Class}
150              * method.</i>
151              * </p>
152              *
153              * @example
154              *   var point = new foodev.bar.Point();
155              *   point.getFullClassName(); // 'foodev.bar.Point'
156              *
157              * @return {String} The full class name.
158              *
159              * @name Class#getFullClassName
160              * @function
161              */
162             newC.prototype.getFullClassName = function () {
163                 return ((!obj.Private) ? (pack.getFullPackageName() + ".") : '') + cname;
164             };
165             newC.getFullClassName = newC.prototype.getFullClassName;
166 
167             return newC;
168         };
169 
170         /**
171          * Returns the name of the package the method is invoked on.
172          *
173          * <p>
174          * <i>This method exist on all packages/subpackages created using the
175          * {@link foodev.Package} and {@link Package#Package} method.</i>
176          * </p>
177          *
178          * @example
179          * foodev.bar.getPackageName(); // 'bar'
180          *
181          * @return {String} The name of the package the method is invoked on.
182          *
183          * @name Package#getPackageName
184          * @function
185          */
186         this.getPackageName = function () {
187             return name;
188         };
189 
190         /**
191          * Returns parent package if one exist, otherwise returns <code>window</code>.
192          *
193          * <p>
194          * <i>This method exist on all packages/subpackages created using the
195          * {@link foodev.Package} method.</i>
196          * </p>
197          *
198          * @return {Package} The owning package of the current package.
199          *
200          * @name Package#getPackageOwner
201          * @function
202          */
203         this.getPackageOwner = function () {
204             return owner;
205         };
206 
207         /**
208          * Returns the full name of the package.
209          *
210          * <p>
211          * <i>This method exist on all packages/subpackages created using the
212          * {@link foodev.Package} method.</i>
213          * </p>
214          *
215          * @example
216          *   foodev.bar.getFullPackageName(); // 'foodev.bar'.
217          *
218          * @return {String} The full name of the current package.
219          *
220          * @name Package#getFullPackageName
221          * @function
222          */
223         this.getFullPackageName = function () {
224             if (owner && owner instanceof Package) {
225                 return owner.getFullPackageName() + '.' + this.getPackageName();
226             } else {
227                 return this.getPackageName();
228             }
229         };
230         if (foodev.debug) {
231             this.toString = this.getFullPackageName;
232         }
233 
234 
235         /**
236          * Creates a new class under the package the function is being invoked on. Every package
237          * created using {@link Package#Package} will have this method for creating classes
238          * under that package.
239          *
240          * <p>
241          * <i>This method exist on all packages/subpackages created using the
242          * {@link foodev.Package} method.</i>
243          * </p>
244          *
245          * @example
246          *   // This constructs a new class at foodev.bar.Test
247          *   foodev.bar.Class( 'Test', {
248          *       initialize: function () {
249          *           // constructor
250          *       }
251          *   });
252          *   new foodev.bar.Test() instanceof foodev.bar.Test // true
253          *
254          *   // This constructs a new class and returns it.
255          *   // This allows a class to be used in a closure and remain "private".
256          *   // foodev.Test will not be available in this example. Instead <code>MyPrivateClass</code>
257          *   // in the current scope/closures will.
258          *   var MyPrivateClass = foodev.Class( 'Test', {
259          *       Private: true,
260          *       initialize: function () {
261          *           // constructor
262          *       }
263          *   });
264          *
265          * @param {String} cname The name of the new class to create.
266          * @param {Object} obj Object containing the functions/fields to put into the new class
267          *          prototype.
268          * @param {Function} obj.initialize A function that is used to construct new instances of
269          *          the class.
270          * @param {Class} [obj.Extends] A class to inherit from.
271          * @param {Boolean} [obj.Private=false] If set to true the class will not be exposed in the
272          *          Package. It will be returned instead allowing for it to be used in a closure
273          *          and thus remain 'private'.
274          *
275          * @name Package#Class
276          * @function
277          */
278         this.Class = function (cname, obj) {
279             var Class = makeClass(this, cname, obj);
280             if (obj.Private) {
281                 return Class;
282             } else {
283                 this[cname] = Class;
284             }
285         };
286 
287         /**
288          * Creates a package underneath this current package. All packages created this way
289          * receives three functions: {@link Package#Class}, {@link Package#Package},
290          * {@link Package#RootPackage} which are used to create classes and further subpackages
291          * under the current package.
292          *
293          * <p>
294          * <i>This method exist on all packages/subpackages created using the
295          * {@link foodev.Package} method.</i>
296          * </p>
297          *
298          * @example
299          *   Create a package under foodev.example
300          *   foodev.Package('example'); // foodev.example is now available
301          *   foodev.Package('example.bar.baz') foodev.example.bar.baz is now available
302          *
303          * @param {String} subPackageName The new package name. This can be dot notation to
304          *      create deep sub packages.
305          *
306          * @name Package#Package
307          * @function
308          */
309         this.Package = function (subPackageName) {
310             var subs = subPackageName.split('.');
311             var i;
312             var cur = this;
313             for (i = 0; i < subs.length; i++) {
314                 var pack = cur[subs[i]];
315                 if (pack && !(pack instanceof Package)) {
316                     pack = null;
317                 }
318                 if (!pack) {
319                     pack = new Package(cur, subs[i]);
320                     cur[subs[i]] = pack;
321                 }
322                 cur = pack;
323             }
324         };
325 
326         /**
327          * Creates a new root package under the window scope.
328          *
329          * @example
330          *   Create a root package named example.
331          *   foodev.RootPackage('example'); // Object example is now avaiable in the global scope.
332          *
333          * @param {String} rootPackageName The root package name.
334          *
335          * @name Package#RootPackage
336          * @function
337          */
338         this.RootPackage = function (rootPackageName) {
339             if (rootPackageName.indexOf('.') >= 0) {
340                 throw new Error("Can't create root package name with dots");
341             }
342             var pack = window[rootPackageName];
343             var prev;
344             if (pack && !(pack instanceof Package)) {
345                 prev = pack;
346                 pack = null;
347             }
348             if (!pack) {
349                 pack = new Package(window, rootPackageName);
350                 if (prev) {
351                     foodev.extend(pack, prev);
352                 }
353                 window[rootPackageName] = pack;
354             }
355         };
356 
357     };

360     /**
361      * Copy all properties of a source object to a target object. Modifies the passed in target
362      * object. Any properties on the source object that are set to undefined will not be (re)set on
363      * the destination object.
364      *
365      * @param {Object} target The object that will be modified or null to make new object.
366      * @param {Object} source The object with properties to be set on the destination. It's
367      *          possible to pass multiple source objects.
368      *
369      * @returns {Object} The destination object. In case target is null, this is a newly created
370      *          object.
371      *
372      * <p>
373      * Function borrowed from OpenLayers: Copyright (c) 2005-2008 MetaCarta, Inc.
374      * </p>
375      */
376     foodev.extend = function (target, source) {
377 
378         target = target || {};
379         var i, len, property;
380 
381         if (arguments.length >= 1) {
382             for (i = 1, len = arguments.length; i < len; i = i + 1) {
383 
384                 source = arguments[i];
385 
386                 if (source) {
387                     for (property in source) {
388                         if (source.hasOwnProperty(property)) { // a bit sily
389                             var value = source[property];
390                             if (value !== undefined) {
391                                 target[property] = value;
392                             }
393                         }
394                     }
395 
396                     /**
397                      * IE doesn't include the toString property when iterating over an object's
398                      * properties with the for (property in object) syntax. Explicitly check if the
399                      * source has its own toString property.
400                      *
401                      * FF/Windows < 2.0.0.13 reports "Illegal operation on WrappedNative prototype
402                      * object" when calling hawOwnProperty if the source object is an instance of
403                      * window.Event.
404                      */
405                     var sourceIsEvt = (typeof window.Event === 'function') &&
406                                       (source instanceof window.Event);
407 
408                     if (!sourceIsEvt && source.hasOwnProperty('toString')) {
409                         target.toString = source.toString;
410                     }
411                 }
412 
413             }
414         }
415 
416         return target;
417 
418     };
