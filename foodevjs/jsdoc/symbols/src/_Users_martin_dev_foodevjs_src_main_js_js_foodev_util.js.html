 12          * Helper method to assert that a required parameter is not undefined and if it has a value, is
 13          * of the specified type given.
 14          *
 15          * <p>It's possible to turn off asserts by setting <code>foodev.assert</code> to false before
 16          * loading the util.js file.</p>
 17          *
 18          * @param {Object} val Value of the parameter. This will be asserted to be a value and
 19          * @param {String} name Name of the parameter, used for throw
 20          * @param {Mixed} [type] Either a string specifying a primitive type such as 'number', 'string',
 21          *          'boolean', 'object', 'function' or a real function to specify an instanceof object
 22          *          type. Primitive type 'array' and (DOM) 'element' are also handled. If not
 23          *          provided, type is not enforced.
 24          * @param {boolean} [undefinedOk=false] If it is okay with undefined values. Default is false.
 25          * @param {boolean} [nullOk=false] If it is okay with null values. Default is false.
 26          */
 27         foodev.assert = function (val, name, type, undefinedOk, nullOk) {
 28             if (val === undefined) {
 29                 if (!undefinedOk) {
 30                     throw new Error('"' + name + '" is undefined');
 31                 }
 32             } else if (val === null) {
 33                 if (!nullOk) {
 34                     throw new Error('"' + name + '" is null');
 35                 }
 36             } else if (type) {
 37                 if (typeof type === 'string') {
 38                     if ((type === 'array' && foodev.arrIs(val)) ||
 39                             (type === 'element' && foodev.elIs(val)) ||
 40                             typeof val === type) {
 41                         return;
 42                     } else {
 43                         throw new Error('"' + name + '" is not of required type: ' + type);
 44                     }
 45                 } else if (typeof type === 'function') {
 46                     if (!(val instanceof type)) {
 47                         var f = type.getFullClassName ? type.getFullClassName() : '?';
 48                         throw new Error('"' + name + '" is not instanceof: ' + f);
 49                     }
 50                 }
 51             }
 52         };

 57     /**
 58      * Helper method for syntesizing <code>equals()</code> methods of objects.
 59      *
 60      * @param type Type to compare instanceof with.
 61      * @return {Function} a function with signature <code>function (other)</code> that compares
 62      *          the incoming 'other' object with <code>this</code>
 63      * @private
 64      */
 65     foodev.makeEquals = function (type) {
 66         return function (other) {
 67             if (!other) {
 68                 return false;
 69             }
 70             if (!(other instanceof type)) {
 71                 return false;
 72             }
 73             var prop;
 74             for (prop in this) {
 75                 if (this.hasOwnProperty(prop)) {
 76                     var v1 = this[prop];
 77                     var v2 = other[prop];
 78                     if (typeof v1 === 'function') {
 79                         continue;
 80                     }
 81                     if (v1 !== v2) {
 82                         return false;
 83                     }
 84                 }
 85             }
 86             return true;
 87         };

448                 complete = foodev.bind(req.bind, complete);

507     foodev.load = function (url, options) {
508

510         foodev.assert(options, 'options', 'object', true);
511         if (options) {
512             foodev.assert(options.type, 'options.type', 'string', true);

521             container = document.getElementsByTagName('head');
522             if (container.length > 0) {
523                 container = container[0];
524             } else {
525                 container = document.body;
526             }

528         case 'image':
529             el = document.createElement('img');
530             el.alt = '';
531             container = document.body;
532             break;
533         }

536         var loaded = false;
537         var fn = function () {
538             var rs = this.readyState || '';
539             if (!loaded && (!rs || rs === "loaded" || rs === "complete" || rs === 'uninitialized')) {
540                 loaded = true;
541 
542                 this.onload = null;
543                 this.onerror = null;
544                 this.onreadystatechange = null;
545 
546                 if (this.parentNode && this.parentNode.removeChild) {

548                         this.parentNode.removeChild(this);
549                     } catch (e) {
550                         // Do nothing
551                     }
552                 }
553 
554                 if (options.error && this.$isError) {
555                     options.error.call(options.bind || this);
556                 } else if (options.success) {
557                     options.success.call(options.bind || this);
558                 }
559             }

566         }

570             this.$isError = true;
571             fn.apply(this, arguments);
572         };
573 
574         if (container) {
575             container.appendChild(el);
576         }
577 
578         if (!el.src) {
579             el.src = url;
580         }
581

845     /**
846      * Measures the given element even if it is not currently attached. The measurement includes,
847      * width, height and absolute position from document top left (0,0).
848      *
849      * @param {Element} el the element to measure.
850      * @param {Element} [attachTo] optional element to attach the element to when measuring.
851      * @return {Object} with the properties <code>w</code> for width and <code>h</code> for
852      *          height in pixels, <code>x</code> for offset x, <code>y</code> for offset y.
853      */
854     foodev.measure = function (el, attachTo) {
855 
856         foodev.assert(el, 'el', 'element');
857         foodev.assert(attachTo, 'attachTo', 'element', true);
858 
859         var tmp = el;
860         var attachNode = null;
861         if (!attachTo) {
862             attachTo = document.body;
863         }
864 
865         if (tmp !== document.body) {
866             while (true) {
867 
868                 if (tmp.parentNode === document.body) {
869                     break;
870                 } else if (!tmp.parentNode) {
871                     attachNode = tmp;
872                     break;
873                 }
874 
875                 tmp = tmp.parentNode;
876 
877             }
878         }
879 
880         try {
881 
882             if (attachNode) {
883 
884                 $measure_setstyle(attachNode, 'top', '0');
885                 $measure_setstyle(attachNode, 'left', '-5000px');
886                 $measure_setstyle(attachNode, 'right', '');
887                 $measure_setstyle(attachNode, 'bottom', '');
888                 $measure_setstyle(attachNode, 'position', 'absolute');
889 
890                 attachTo.appendChild(attachNode);
891 
892             }
893 
894             var x = 0;
895             var y = 0;
896 
897             tmp = el;
898 
899             if (!attachNode) {
900 
901                 if (tmp !== document.body) {
902                     while (true) {
903 
904                         x += tmp.offsetLeft;
905                         y += tmp.offsetTop;
906 
907                         if (tmp.offsetParent === document.body) {
908                             break;
909                         }
910 
911                         tmp = tmp.offsetParent;
912 
913                     }
914                 }
915 
916             }
917 
918             return {
919                 w: el.offsetWidth,
920                 h: el.offsetHeight,
921                 x: x,
922                 y: y
923             };
924 
925         } finally {
926 
927             if (attachNode) {
928                 attachNode.parentNode.removeChild(attachNode);
929                 $measure_restore(attachNode, 'top');
930                 $measure_restore(attachNode, 'left');
931                 $measure_restore(attachNode, 'right');
932                 $measure_restore(attachNode, 'bottom');
933                 $measure_restore(attachNode, 'position');

940     /**
941      * Turns the given object of key-values into a string of request parameters. Will not prepend an ampersand (&)
942      * or question mark (?) in front of the first value. Handles also the case where the value is an array by turning
943      * the values into a comma separated string.
944      * <code><pre>
945      * var obj = {
946      *    key1: 'value1',
947      *    key2: 'value2',
948      *    key3: ['a1', 'a2', 'a3']
949      * };
950      *
951      * var str = foodev.objToReqParams(obj);
952      *
953      * // str is now 'key1=value1&key2=value2&key3=a1,a2,a3'
954      * </pre><code>
955      *
956      * @param {String} param the parameter object to turn into request parameter string.
957      * @return the request parameter string.
958      */
959     foodev.objToReqParams = function (param) {
960 
961         foodev.assert(param, 'param', 'object');
962 
963         var result = [];
964         var key, val, i, n, item;
965 
966         for (key in param) {
967             if (param.hasOwnProperty(key)) {
968 
969                 val = param[key];
970 
971                 if (val !== null) {
972 
973                     if (foodev.arrIs(val)) {
974 
975                         var arr = [];
976 
977                         for (i = 0, n = val.length; i < n; i++) {
978                             item = val[i];
979                             arr.push(encodeURIComponent(item !== null && item !== undefined ? "" : item));
980                         }
981 
982                         val = arr.join(',');
983 
984                     } else {
985 
986                         val = encodeURIComponent(val);
987 
988                     }
989 
990                     result.push(encodeURIComponent(key) + '=' + val);
991 
992                 }
993 
994             }
995 
996         }
997 
998         return result.join('&');
