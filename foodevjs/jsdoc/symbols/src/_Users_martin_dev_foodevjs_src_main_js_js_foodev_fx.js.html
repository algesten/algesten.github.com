  5     var $tween_number = function (arg1, arg2) {
  6         if (typeof arg1 !== 'number' || typeof arg2 !== 'number') {
  7             return;
  8         }
  9         return {
 10             from: arg1,
 11             to: arg2
 12         };
 13     };
 14 
 15     var $tween_string = function (arg1, arg2) {
 16         if (typeof arg1 !== 'string' || typeof arg2 !== 'string') {
 17             return;
 18         }
 19 
 20         if (arg1.length < 3 || arg2.length < 3) {
 21             return;
 22         }
 23 
 24         var suffix = arg1.substring(arg1.length - 2);
 25 
 26         if (arg2.substring(arg2.length - 2) !== suffix) {
 27             return;
 28         }
 29 
 30         var val1 = parseFloat(arg1);
 31         var val2 = parseFloat(arg2);
 32 
 33         if (!isFinite(val1) || !isFinite(val2)) {
 34             return;
 35         }
 36 
 37         return {
 38             from: val1,
 39             to: val2,
 40             suffix: suffix
 41         };
 42 
 43     };
 44 
 45     var $tween_buildTweenProp = function (prop, from, to) {
 46 
 47         var propFrom = from[prop];
 48         var propTo = to[prop];
 49 
 50         var tweenProp = $tween_number(propFrom, propTo);
 51 
 52         if (!tweenProp) {
 53             tweenProp = $tween_string(propFrom, propTo);
 54         }
 55 
 56         if (!tweenProp) {
 57             tweenProp = {
 58                 from: propFrom,
 59                 to: propTo,
 60                 notween: true
 61             };
 62         }
 63 
 64         tweenProp.prop = prop;
 65 
 66         return tweenProp;
 67 
 68     };
 69 
 70     var $tween_buildTweenProps = function (tween, from, to) {
 71 
 72         var prop;
 73 
 74         for (prop in from) {
 75 
 76             if (from.hasOwnProperty(prop)) {
 77 
 78                 tween.props.push($tween_buildTweenProp(prop, from, to));
 79 
 80             }
 81 
 82         }
 83 
 84         for (prop in to) {
 85 
 86             if (to.hasOwnProperty(prop) && !from.hasOwnProperty(prop)) {
 87 
 88                 tween.props.push($tween_buildTweenProp(prop, from, to));
 89 
 90             }
 91 
 92         }
 93 
 94     };

 97     /**
 98      * Tweens properties between (numerical) values.
 99      * 
100      * @param {Number} duration time in milliseconds for tween.
101      * @param {Element} el the node to change style attributes on
102      * @param {Object} from definition of properties to tween from.
103      * @param {Object} from.prop start value for property named 'prop'. Valid values are numerics
104      *            and strings that holds number suffixed with 'px' or 'em'.
105      * @param {Object} [from.onStart] function to invoke upon start.
106      * @param {Object} [from.onEnd] function to invoke upon completion.
107      * @param {Object} to definition of properties to tween to.
108      * @param {Object} to.prop end value for property named 'prop'.
109      * @return {Object} with one single function <code>cancel()</code> which can be used to cancel
110      *         animation.
111      */
112     foodev.tween = function (duration, el, from, to) {
113 
114         foodev.assert(duration, 'duration', 'number');
115         foodev.assert(el, 'el', 'element');
116         foodev.assert(from, 'from', 'object');
117         foodev.assert(to, 'to', 'object');
118 
119         var tweens = [];
120 
121         var i;
122         for (i = 1; i < arguments.length; i += 3) {
123             foodev.assert(arguments[i], 'el' + i.toString(), 'element');
124             foodev.assert(arguments[i + 1], 'from' + i.toString(), 'object');
125             foodev.assert(arguments[i + 2], 'to' + i.toString(), 'object');
126 
127             var tween = {
128                 el: arguments[i],
129                 onStart: arguments[i + 1].onStart,
130                 onEnd: arguments[i + 1].onEnd,
131                 props: []
132             };
133             tweens.push(tween);
134 
135             $tween_buildTweenProps(tween, arguments[i + 1], arguments[i + 2]);
136 
137         }
138 
139         var startTime = Date.now();
140         var progress;
141         var timerId;
142         var first = true;
143         var cancel = false;
144 
145         var update = function () {
146 
147             // avoid multiple onEnd
148             if (progress === 1) {
149                 return;
150             }
151 
152             if (cancel) {
153                 progress = 1;
154             } else {
155                 progress = (Date.now() - startTime) / duration;
156             }
157 
158             // avoid multiple onStart
159             if (progress === 0 && !first) {
160                 return;
161             }
162 
163             if (first) {
164 
165                 progress = 0;
166                 first = false;
167 
168             } else if (progress >= 1) {
169 
170                 clearInterval(timerId);
171                 progress = 1;
172 
173             }
174 
175             for (i = 0; i < tweens.length; i++) {
176 
177                 if (progress === 0) {
178                     if (tweens[i].onStart) {
179                         tweens[i].onStart(tweens[i].el);
180                     }
181                 }
182 
183             }
184 
185             for (i = 0; i < tweens.length; i++) {
186 
187                 var j;
188                 var props = tweens[i].props;
189                 for (j = 0; j < props.length; j++) {
190 
191                     if (props[j].notween) {
192 
193                         if (progress === 0 && props[j].from) {
194 
195                             tweens[i].el.style[props[j].prop] = props[j].from;
196 
197                         } else if (progress === 1 && props[j].to) {
198 
199                             tweens[i].el.style[props[j].prop] = props[j].to;
200 
201                         }
202 
203                     } else {
204 
205                         var scale = (1 - Math.cos(Math.PI * progress)) / 2;
206 
207                         var val = props[j].from + (props[j].to - props[j].from) * scale;
208 
209                         if (props[j].suffix) {
210                             val = val.toString() + props[j].suffix;
211                         }
212 
213                         tweens[i].el.style[props[j].prop] = val;
214 
215                     }
216 
217                 }
218 
219             }
220 
221             for (i = 0; i < tweens.length; i++) {
222 
223                 if (progress === 1) {
224                     if (tweens[i].onEnd) {
225                         tweens[i].onEnd(tweens[i].el);
226                     }
227                 }
228 
229             }
230 
231         };
232 
233         update();
234 
235         timerId = setInterval(update, 20);
236 
237         return {
238             cancel: function () {
239                 cancel = true;
240                 update();
241             }
242         };
243 
244     };
